Description

Constraint Rule Popover Component.

Creates a popover on an element based on the constraint rule list and product passed to it.

Implements

OnChanges

Example

<button
[popover]="popTemplate"
container="body"
[outsideClick]="true"
containerClass="CRPopover"
triggers="mouseenter:mouseleave focus:blur"
>
Add to Cart
</button>

<ng-template #popTemplate>
<apt-cr-popover [ruleList]="constraintRuleList" [product]="product"></apt-cr-popover>
</ng-template>

<div class="p-2">
  <div class="popoverHeader mb-3">
    <h5>Product(s) Included / Excluded</h5>
  </div>
  <div class="popoverBody">
    <div *ngIf="message">
      <p [innerHTML]="message"></p>
    </div>

    <div *ngIf="productInclusionRules?.length > 0">
      <p>
        <strong>The following products </strong> are included with the product <strong> {{product.Name}}</strong>.
      </p>
      <div class="border-top pt-3">
        <div *ngFor="let rule of productInclusionRules" class="d-flex justify-content-between">
          <ng-container *ngIf="rule.ProductScope === 'Product'">
            <a [routerLink]="['/product', rule.Product[productService.config.productIdentifier]]" href="javascript:void(0)">{{rule?.Product?.Name}}</a>
            <small>
              <apt-price [record]="rule.Product"></apt-price>
            </small>
          </ng-container>
        </div>
      </div>
    </div>

    <div *ngIf="productExclusionRules?.length > 0">
      <p class="border-top pt-3 mt-3">
        <strong>The following products </strong> are excluded with the product <strong> {{product.Name}}</strong>.
      </p>
      <div class="border-top pt-3">
        <div *ngFor="let rule of productExclusionRules" class="d-flex justify-content-between">
          <a [routerLink]="['/product', rule.Product[productService.config.productIdentifier]]" href="javascript:void(0)">{{rule?.Product?.Name}}</a>
          <small>
            <apt-price [record]="rule.Product"></apt-price>
          </small>
        </div>
      </div>
    </div>

    <div *ngIf="otherRules?.length > 0">
      <p *ngFor="let rule of otherRules" [innerHTML]="rule.Message"></p>
    </div>
  </div>
</div>
